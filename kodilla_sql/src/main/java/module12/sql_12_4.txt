select ISL.NAME, ISS.DESCRIPTION
from ISSUESLISTS ISL, ISSUES ISS
where ISL.ID = ISS.ISSUELIST_ID;


select U.FIRSTNAME, U.LASTNAME, ISS.DESCRIPTION
from USERS U, ISSUES ISS
where U.ID = ISS.USER_ID_ASSIGNEDTO;


select U.FIRSTNAME, U.LASTNAME, count(*) as ISSUES_QTY
from USERS U, ISSUES ISS
where U.ID = ISS.USER_ID_ASSIGNEDTO
group by U.LASTNAME
having count(*) > 0;
